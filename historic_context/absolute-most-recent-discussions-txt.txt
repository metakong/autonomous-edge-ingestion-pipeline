# src/governance.py
from enum import Enum
from functools import wraps
from typing import Any, Callable, Dict, List, Optional
from pydantic import BaseModel, Field

class DSIEStage(str, Enum):
    DIAGNOSE = "DSIE-01"
    STRATEGIZE = "DSIE-02"
    IMPLEMENT = "DSIE-03"
    EXECUTE = "DSIE-04"

class DiagnosisReport(BaseModel):
    stage: DSIEStage = Field(default=DSIEStage.DIAGNOSE)
    problem_summary: str
    evidence: Dict[str, Any] = Field(default_factory=dict)
    root_cause_hypothesis: str
    confidence: float = Field(ge=0.0, le=1.0)
    alternatives_considered: List[str] = Field(default_factory=list)

class StrategyOption(BaseModel):
    name: str
    category: str 
    description: str
    expected_benefit: str
    risks: List[str] = Field(default_factory=list)
    kill_switch_conditions: List[str] = Field(default_factory=list)

class StrategyObject(BaseModel):
    stage: DSIEStage = Field(default=DSIEStage.STRATEGIZE)
    problem_summary: str
    options: List[StrategyOption]
    recommended_option: str
    rationale: str

class ExecutionReport(BaseModel):
    stage: DSIEStage = Field(default=DSIEStage.EXECUTE)
    subsystem: str
    change_summary: str
    primary_metric: str
    metric_before: float
    metric_after: float
    observation_window_hours: float
    success: bool
    notes: str = ""

class AgentContract(BaseModel):
    agent_id: str
    human_readable_name: str
    autonomy_level: int
    primary_metric: Optional[str] = None

class AgentContext(BaseModel):
    contract: AgentContract
    current_diagnosis: Optional[DiagnosisReport] = None
    current_strategy: Optional[StrategyObject] = None

class GovernanceError(RuntimeError):
    pass

def require_diagnosis(fn: Callable) -> Callable:
    @wraps(fn)
    def wrapper(*args, **kwargs):
        ctx: Optional[AgentContext] = kwargs.get("ctx")
        if ctx is None:
            raise GovernanceError("AgentContext 'ctx' is required.")
        if ctx.current_diagnosis is None:
            raise GovernanceError("DSIE-01 DiagnosisReport required before calling this function.")
        return fn(*args, **kwargs)
    return wrapper

def require_strategy(fn: Callable) -> Callable:
    @wraps(fn)
    def wrapper(*args, **kwargs):
        ctx: Optional[AgentContext] = kwargs.get("ctx")
        if ctx is None:
            raise GovernanceError("AgentContext 'ctx' is required.")
        if ctx.current_strategy is None:
            raise GovernanceError("DSIE-02 StrategyObject required before calling this function.")
        return fn(*args, **kwargs)
    return wrapper

def perform_live_diagnosis() -> DiagnosisReport:
    """
    Probes the target to generate REAL evidence for the Diagnosis Report.
    """
    logger.info("Starting DSIE-01 Diagnostic Probe...")
    
    # 1. Probe
    start_time = time.time()
    try:
        # We use a lightweight request (or the real one if API doesn't support HEAD)
        response = requests.get(URL, headers=HEADERS, params=PARAMS, timeout=5)
        latency = (time.time() - start_time) * 1000
        
        # 2. Gather Evidence
        evidence = {
            "timestamp": datetime.utcnow().isoformat(),
            "target_url": URL,
            "status_code": response.status_code,
            "latency_ms": round(latency, 2),
            "content_type": response.headers.get("Content-Type", "unknown"),
            "content_length": len(response.content)
        }
        
        # 3. Formulate Hypothesis
        if response.status_code == 200 and "application/json" in evidence["content_type"]:
            hypothesis = "Target API is healthy and accessible."
            confidence = 0.99
        elif response.status_code == 403:
            hypothesis = "Target is blocking requests (WAF/Cloudflare)."
            confidence = 0.95
        else:
            hypothesis = f"Target returned unexpected status: {response.status_code}"
            confidence = 0.8
            
        # 4. Create Report
        report = DiagnosisReport(
            stage="DSIE-01",
            problem_summary="Scheduled Data Ingestion",
            evidence=evidence,
            root_cause_hypothesis=hypothesis,
            confidence=confidence,
            alternatives_considered=["Browser Scraping (Slower)", "Sporttrade (Unstable)"]
        )
        
        # 5. RECORD THE REPORT (SOP Requirement)
        # We save it to disk so the Auditor can review it later
        report_path = f"artifacts/DSIE-01_dk_{int(time.time())}.json"
