            # 1. Find the Folder ID
            query = f"name = '{folder_name}' and mimeType = 'application/vnd.google-apps.folder' and trashed = false"
            results = self.service.files().list(q=query, fields="files(id, name)").execute()
            items = results.get('files', [])
            
            if not items:
                logger.warning(f"Folder '{folder_name}' not found. Creating it.")
                file_metadata = {
                    'name': folder_name,
                    'mimeType': 'application/vnd.google-apps.folder'
                }
                folder = self.service.files().create(body=file_metadata, fields='id').execute()
                folder_id = folder.get('id')
            else:
                folder_id = items[0]['id']

            # 2. Upload the File
            file_metadata = {
                'name': filename,
                'parents': [folder_id]
            }
            
            json_str = json.dumps(data, indent=2)
            media = MediaIoBaseUpload(io.BytesIO(json_str.encode('utf-8')), mimetype='application/json')
            
            file = self.service.files().create(
                body=file_metadata,
                media_body=media,
                fields='id'
            ).execute()
            
            logger.info(f"File ID: {file.get('id')} uploaded to Drive.")
            return True
        except Exception as e:
            logger.error(f"Drive Upload Failed: {e}")
            return False

from playwright.async_api import async_playwright
import logging

logger = logging.getLogger("VeiledVector")

async def scrape_url(url: str):
    """
    Scrapes a URL using Playwright via the local SOCKS5 proxy (The Dell).
    """
    logger.info(f"Scraping {url} via Stealth Tunnel...")
    async with async_playwright() as p:
        # Proxy is set to the local Tailscale tunnel
        browser = await p.chromium.launch(
            proxy={"server": "socks5://localhost:1055"}
        )
        context = await browser.new_context(
            user_agent="Mozilla/5.0 (Windows NT 10.0; Win64; x64) AppleWebKit/537.36 (KHTML, like Gecko) Chrome/119.0.0.0 Safari/537.36"
        )
        page = await context.new_page()
        
        try:
            await page.goto(url, timeout=60000, wait_until="domcontentloaded")
            # Wait for dynamic content if needed (simple wait for now)
            await page.wait_for_timeout(5000) 
            
            content = await page.content()
            title = await page.title()
            
            await browser.close()
            return {"title": title, "html": content[:100000]} # Truncate to save tokens
        except Exception as e:
            await browser.close()
            logger.error(f"Scrape Failed: {e}")
            raise e

import logging
import google.generativeai as genai
import os
import json
import datetime
from src.schemas import DiagnosisReport, StrategyOption, ExecutionReport
from src.tools.scraper import scrape_url
from src.tools.drive import DriveClient

logging.basicConfig(level=logging.INFO)
logger = logging.getLogger("VeiledVector")

class DSIECore:
    def __init__(self):
        self.api_key = os.getenv("GOOGLE_AI_STUDIO_KEY")
        if self.api_key:
            genai.configure(api_key=self.api_key)
            self.model = genai.GenerativeModel('gemini-1.5-pro')
        
        # Initialize Drive Client lazily or on demand to handle startup errors gracefully
        self.drive = None

    async def run_cycle(self, task: str):
        """
        The Production DSIE Loop.
        1. DIAGNOSE: Understand the task (Extract URL).
        2. IMPLEMENT: Scrape the data via Stealth Proxy.
        3. STRATEGIZE: Parse data with Gemini.
        4. EXECUTE: Save to Drive.
        """
        logger.info(f"STARTING MISSION: {task}")
        
        # 1. EXTRACT URL (Simple Heuristic for Pilot)
        # In full version, Gemini extracts this. For pilot, we look for 'http'
        target_url = next((word for word in task.split() if word.startswith("http")), None)
        if not target_url:
            if "DraftKings" in task: target_url = "https://sportsbook.draftkings.com/"
            else: return {"status": "Failed", "error": "No URL found in task"}

        # 2. SCRAPE (The Hands)
        try:
            scrape_data = await scrape_url(target_url)
        except Exception as e:
            return {"status": "Scrape Failed", "error": str(e)}

        # 3. ANALYZE (The Brain)
        try:
            prompt = f"""
            Analyze this HTML from {target_url}. 
            Task: {task}
            Extract the relevant odds or data points into valid JSON.
            HTML Context: {scrape_data['html'][:20000]}
            """
            response = self.model.generate_content(prompt)
            # Clean markdown code blocks if present
            clean_json = response.text.replace('```json', '').replace('```', '')
            parsed_data = json.loads(clean_json)
        except Exception as e:
            logger.error(f"Gemini Analysis Failed: {e}")
            parsed_data = {"raw_title": scrape_data['title'], "error": "AI Parsing Failed"}

        # 4. SAVE (The Memory)
        try:
            if not self.drive:
                self.drive = DriveClient()
            
            filename = f"mission_report_{datetime.datetime.now().strftime('%Y%m%d_%H%M%S')}.json"
            self.drive.upload_json(filename, parsed_data)
            return {"status": "Mission Complete", "file": filename, "data_preview": parsed_data}
        except Exception as e:
            return {"status": "Upload Failed", "error": str(e)}
EOF
