import json
from datetime import datetime

class GovernanceLayer:
    """
    Enforces Constitution-aligned governance gates before execution.
    """
    
    def check_escalation_triggers(self, decision):
        """
        Returns (is_safe: bool, reason: str)
        """
        checks = [
            ("confidence_below_80", decision.confidence < 0.80),
            ("tool_failure_count_exceeded", decision.tool_failures >= 2),
            ("data_integrity_failed", decision.data_integrity_score < 0.80),
            ("ambiguous_instruction", decision.instruction_clarity < 0.90),
            ("prompt_injection_detected", self._detect_prompt_injection(decision)),
        ]
        
        for check_name, is_triggered in checks:
            if is_triggered:
                return False, f"ESCALATE: {check_name}"
        
        return True, "APPROVED"
    
    def _detect_prompt_injection(self, decision):
        """
        Quick heuristic: Look for instruction-like patterns in data.
        """
        suspicious_patterns = [
            "ignore previous",
            "override",
            "new instructions",
            "forget about",
            "system prompt"
        ]
        
        text = str(decision).lower()
        return any(pattern in text for pattern in suspicious_patterns)
    
    def audit_log(self, mission_id, decision, approval):
        """
        Persistent audit trail for compliance/debugging.
        """
        log_entry = {
            "timestamp": datetime.utcnow().isoformat(),
            "mission_id": mission_id,
            "decision_summary": str(decision)[:500],
            "governance_approval": approval,
            "escalation_reason": approval.get("reason", ""),
        }
        
        # TODO: Write to Firestore collection "audit_log"
        print(json.dumps(log_entry, indent=2))
        
        return log_entry
